    const calGrid = document.getElementById("calGrid");
    const dayTitle = document.getElementById("dayTitle");
    const dayMeta = document.getElementById("dayMeta");
    const routineList = document.getElementById("routineList");
    const heavyTaskEl = document.getElementById("heavyTask");
    const morningStory = document.getElementById("morningStory");
    const eveningStory = document.getElementById("eveningStory");
    const doneBtn = document.getElementById("doneBtn");
    const undoneBtn = document.getElementById("undoneBtn");
    const statusNote = document.getElementById("statusNote");
    const resetBtn = document.getElementById("resetBtn");
    const todayBtn = document.getElementById("todayBtn");
    const hideDoneToggle = document.getElementById("hideDoneToggle");

    let selectedDay = null;
    let doneSet = loadDoneSet();

    function loadDoneSet(){
      try{
        const raw = localStorage.getItem(STORAGE_KEY);
        if(!raw) return new Set();
        const arr = JSON.parse(raw);
        return new Set(arr);
      }catch(e){
        return new Set();
      }
    }

    function saveDoneSet(){
      localStorage.setItem(STORAGE_KEY, JSON.stringify(Array.from(doneSet)));
    }

    function renderCalendar(){
      calGrid.innerHTML = "";
      const hideDone = hideDoneToggle.checked;

      for(let d=1; d<=30; d++){
        const isDone = doneSet.has(d);
        if(hideDone && isDone) continue;

        const btn = document.createElement("button");
        btn.type = "button";
        btn.className = "dayBtn" + (isDone ? " done" : "") + (selectedDay===d ? " active" : "");
        btn.dataset.day = String(d);

        const dot = document.createElement("span");
        dot.className = "dot";
        btn.appendChild(dot);

        const n = document.createElement("div");
        n.className = "dayNum";
        n.textContent = d;
        btn.appendChild(n);

        const m = document.createElement("div");
        m.className = "dayMeta";
        m.textContent = weekLabel(d);
        btn.appendChild(m);

        btn.addEventListener("click", () => selectDay(d));
        calGrid.appendChild(btn);
      }
    }

    function weekLabel(day){
      if(day<=7) return "Week 1";
      if(day<=14) return "Week 2";
      if(day<=21) return "Week 3";
      return "Week 4";
    }

    function selectDay(day){
      selectedDay = day;
      const entry = stories.find(s => s.day === day);

      dayMeta.textContent = `Dag ${day} • ${weekLabel(day)}`;
      dayTitle.textContent = "Wudanbegan";

      routineList.innerHTML = "";
      baseRoutine.forEach(item=>{
        const li = document.createElement("li");
        li.textContent = item;
        routineList.appendChild(li);
      });

      heavyTaskEl.textContent = `Zware taak: ${heavyTasks[day-1]}`;

      morningStory.textContent = entry ? entry.morning : "Geen tekst gevonden.";
      eveningStory.textContent = entry ? entry.evening : "Geen tekst gevonden.";

      doneBtn.disabled = false;
      undoneBtn.disabled = false;

      updateStatusNote();
      renderCalendar();
    }

    function updateStatusNote(){
      if(selectedDay == null){
        statusNote.textContent = "";
        return;
      }
      const isDone = doneSet.has(selectedDay);
      statusNote.textContent = isDone
        ? "Status: afgerond. Je hebt deze dag gemarkeerd als gedaan."
        : "Status: open. Lees, voer uit, markeer als gedaan.";
      doneBtn.textContent = isDone ? "Afgerond" : "Dag gedaan";
      doneBtn.disabled = isDone;
      undoneBtn.disabled = !isDone;
    }

    doneBtn.addEventListener("click", ()=>{
      if(selectedDay == null) return;
      doneSet.add(selectedDay);
      saveDoneSet();
      updateStatusNote();
      renderCalendar();
    });

    undoneBtn.addEventListener("click", ()=>{
      if(selectedDay == null) return;
      doneSet.delete(selectedDay);
      saveDoneSet();
      updateStatusNote();
      renderCalendar();
    });

    resetBtn.addEventListener("click", ()=>{
      doneSet = new Set();
      saveDoneSet();
      renderCalendar();
      updateStatusNote();
    });

    todayBtn.addEventListener("click", ()=>{
      for(let d=1; d<=30; d++){
        if(!doneSet.has(d)){
          selectDay(d);
          return;
        }
      }
      selectDay(30);
    });

    hideDoneToggle.addEventListener("change", ()=>{
      renderCalendar();
    });

    renderCalendar();
    selectDay(1);
  </script>
